function StdEditorDialog(Dialog: Integer; Info: Pointer): Word;
var R: TRect; T: TPoint;
Begin
 case Dialog of
 edReplace: StdEditorDialog :=
Application^.ExecuteDialog(CreateReplaceDialog,Info);
 edReplacePrompt:
    Begin
        R.Assign(0, 1, 40, 8);
        if TPoint.Y <= T.Y then R.Move(0, Desktop.Size.Y);
        StdEditorDialog := MessageBoxRect(R, 'Replace this occurence?',nil,
        mfYesNoCancel + mfInformation);
    End;
End;
